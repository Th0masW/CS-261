The actual script listens for which button is pressed. From there it adds to the speed, or decreses it.
Once the circle drifts out of the container, then the positon is reset to -900 to allow it to pass through the container again. Then to animate it , the script calls itself again.

var button = document.querySelector("#slower");
button.addEventListener("click", slow, false);

var xPos = 0;

function slow() {
xPos -= 5;

circle.style.transform = `translate3d(${xPos}px, 0, 0)`;

if (Math.abs(xPos) >= 900) {
xPos = -500;
}
requestAnimationFrame(slow);
}

function animate() {
xPos += 10;

circle.style.transform = `translate3d(${xPos}px, 0, 0)`;

if (Math.abs(xPos) >= 900) {
xPos = -500;
}
requestAnimationFrame(animate);
}
animate();
