JSON Parse, Stringify

Script to create Object from JSON string:
var text = '{"Characters":[' +
'{"firstName":"Jim","lastName":"Raynor" },' +
'{"firstName":"Sarah","lastName":"Kerrigan" },' +
'{"firstName":"Hierarch","lastName":"Artanis" }]}';

obj = JSON.parse(text);
document.getElementById("JSON-Demo").innerHTML =
obj.Characters[1].firstName + " " + obj.Characters[1].lastName;

Script to create JSON string using Stringify:
var arr = [ "Race":"Zerg", "Race":"Terran", "Race":"Protoss"];
var myJSON = JSON.stringify(arr);
document.getElementById("Stringify").innerHTML = myJSON;
